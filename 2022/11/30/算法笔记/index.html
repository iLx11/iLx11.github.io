int search(int &a, target) {
    int length = sizeof(a) / sizeof(a[0]);
    int mid, left = 0, right = length - 1;
    //[left, right] -- [left, right) -> left<right
    while(left <= right) {
        mid = left + ((right - left) >> 1);
        if(a[mid] > target) {
            //闭区间
            right = mid - 1;
        }else if(a[mid] < target) {
            //闭区间
            left = mid + 1;
        }else if(a[mid] == target) {
            return mid;
        }
    }
    return -1;
}

int remove(int& a, target) {
    int length = sizeof(a) / sizeof(a[0]);
    for(int i = 0; i < length; i++) {
        if(a[i] == target) {
            for(int j = i + 1; j < length; j++) {
                a[j - 1] = a[j];
            }
            i--;
            length--;
        }
    }
    return length;
}

int* dou(int& a) {
    int length = sizeof(a) / sizeof(a[0]);
    int left = 0, right = length - 1, k = length;
    int newArr[length]{};
    for(int i = 0; i < length; i ++) {
        if(a[left] * a[left] < a[right] * a[right]) {
            newArr[k --] =  a[right] * a[right];
            right --;
        }else {
            newArr[k --] =  a[left] * a[left];
            left --;
        }
        a = newArr;
        return a;
    }
}

int minSubArrayLen(int s, int& a) {
    int len = sizeof(a) / sizeof(a[0]);
    int result{INT32_MAX};
    int sum{0}, subLength{0};
    for(int i = 0; i < len; i ++) {
        sum = 0; 
        for(int j = i; j < len; j ++) {
            sum += a[j];
            if(sum >= s) {
                subLength = j - i + 1;
                result = result < subLength ? result : subLength;
                break; //执行完退出
            }
        }
    }
    return result == IN32_MAX ? 0 : result; 
}
滑动窗口解法
int minSubArrayLen(int s, int& a) {
    int result = IN32_MAX;
    int len = sizeof(a) / sizeof(a[0]);
    int i{0}, subLength{0}, sum{0};
    for(int j = 0; j < len; j ++) {
        sum += a[j];
        while(sum >= s) {
            subLength = j - i + 1;
            result = result < subLength ? result : subLength;
            sum -= a[i];
            i ++;
        }
    }
    return result == IN32_MAX ? 0 : result;
}

void generateMatrix(int n, int *array) {
    int loop {0};
    int startx {0};
    int starty {0};
    int mid {n / 2};
    int offset {1};
    int count {1};
    while(loop ++ < n / 2) {
        i = startx;
        j = starty;
        
        //从左往右
        for(int i = startx; i < n - offset; i ++) {
            array[startx][i] = count ++;
        }
        //从上往下
        for(int j = starty; j < n - offset; j ++) {
            array[i][j] = count ++;
        }
        //从右往左
        for(; i > startx; i --) {
            array[i][j] = count ++;
        }
           //从下往上
        for(; j > starty; j --) {
            array[i][j] = count ++;
        }
        startx ++;
        starty ++;
        //收缩1
        offset ++;
    }
    if(n % 2) {
        array[mid][mid] = count;
    }
}
